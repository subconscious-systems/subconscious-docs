```python Python
from subconscious import Subconscious, tools
from pydantic import BaseModel

client = Subconscious(api_key="YOUR_API_KEY")

class WebSearchArgs(BaseModel):
    query: str

run = client.run_and_wait(
    engine="tim",
    input={
        "instructions": "Find the latest news about Tesla's stock performance this week.",
        "tools": [
            tools.function(
                name="web_search",
                description="Search the web for current information",
                url="TOOL_SERVER_URL",
                method="GET",
                timeout=30,
                parameters=WebSearchArgs,
            ),
        ],
    },
)

print(run.result.answer)
```

```javascript JavaScript
import { Subconscious, tools } from "subconscious";
import { z } from "zod";

const client = new Subconscious({
  apiKey: "YOUR_API_KEY",
});

const run = await client.runAndWait({
  engine: "tim",
  input: {
    instructions:
      "Find the latest news about Tesla's stock performance this week.",
    tools: [
      tools.function({
        name: "web_search",
        description: "Search the web for current information",
        url: "TOOL_SERVER_URL",
        method: "GET",
        timeout: 30,
        parameters: z.object({
          query: z.string().describe("The search query"),
        }),
      }),
    ],
  },
});

console.log(run.result?.answer);
```
