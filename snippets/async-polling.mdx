from subconscious import Subconscious

client = Subconscious(api_key="YOUR_API_KEY")

# 1) Start a run without waiting
run = client.run(
    engine="tim-gpt",
    input={
        "instructions": "Run this as an async job",
    },
)

run_id = run.run_id

# Use client.wait() for automatic polling
result = client.wait(
    run_id,
    options={
        "interval_ms": 2000,  # Poll every 2 seconds
        "max_attempts": 60,   # Give up after 60 attempts
    },
)
print(result.result.answer)

```javascript JavaScript
import { Subconscious } from "subconscious";

const client = new Subconscious({
  apiKey: "YOUR_API_KEY",
});

// 1) Start a run without waiting
const run = await client.run({
  engine: "tim-gpt",
  input: {
    instructions: "Run this as an async job",
  },
});

const runId = run.runId;

// 2) Poll for status (manual)
async function pollStatus() {
  while (true) {
    const status = await client.get(runId);
    if (status.status === "succeeded" || status.status === "failed") {
      console.log(status);
      break;
    }
    await new Promise((r) => setTimeout(r, 2000));
  }
}

await pollStatus();

// Or use client.wait() for automatic polling
const result = await client.wait(runId, {
  intervalMs: 2000, // Poll every 2 seconds
  maxAttempts: 60, // Give up after 60 attempts
});
console.log(result.result?.answer);
```
