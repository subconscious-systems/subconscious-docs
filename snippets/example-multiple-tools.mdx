```python Python
from subconscious import Subconscious, tools
from pydantic import BaseModel

client = Subconscious(api_key="YOUR_API_KEY")

class WebSearchArgs(BaseModel):
    query: str

class CalculateArgs(BaseModel):
    expression: str

class SendEmailArgs(BaseModel):
    recipient: str
    subject: str
    body: str

run = client.run_and_wait(
    engine="tim",
    input={
        "instructions": "Research the current Bitcoin price, calculate the percentage change from last week, and email me a summary report.",
        "tools": [
            tools.function(
                name="web_search",
                description="Search the web for current information",
                url="TOOL_SERVER_URL",
                parameters=WebSearchArgs,
            ),
            tools.function(
                name="calculate",
                description="Perform mathematical calculations",
                url="TOOL_SERVER_URL",
                parameters=CalculateArgs,
            ),
            tools.function(
                name="send_email",
                description="Send an email notification",
                url="TOOL_SERVER_URL",
                parameters=SendEmailArgs,
            ),
        ],
    },
)

print(run.result.answer)
```

```javascript JavaScript
import { Subconscious, tools } from "subconscious";
import { z } from "zod";

const client = new Subconscious({
  apiKey: "YOUR_API_KEY",
});

const run = await client.runAndWait({
  engine: "tim",
  input: {
    instructions:
      "Research the current Bitcoin price, calculate the percentage change from last week, and email me a summary report.",
    tools: [
      tools.function({
        name: "web_search",
        description: "Search the web for current information",
        url: "TOOL_SERVER_URL",
        parameters: z.object({
          query: z.string().describe("The search query"),
        }),
      }),
      tools.function({
        name: "calculate",
        description: "Perform mathematical calculations",
        url: "TOOL_SERVER_URL",
        parameters: z.object({
          expression: z.string().describe("Mathematical expression to evaluate"),
        }),
      }),
      tools.function({
        name: "send_email",
        description: "Send an email notification",
        url: "TOOL_SERVER_URL",
        parameters: z.object({
          recipient: z.string().describe("Email recipient"),
          subject: z.string().describe("Email subject"),
          body: z.string().describe("Email body"),
        }),
      }),
    ],
  },
});

console.log(run.result?.answer);
```
