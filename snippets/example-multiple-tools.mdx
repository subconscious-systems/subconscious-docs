                "timeout": 30,
                "parameters": {
                    "type": "object",

                        "expression": {"type": "string", "description": "Mathematical expression to evaluate"}
                    },
                    "required": ["expression"],
                    "additionalProperties": False
                }
            },
            {
                "type": "function",
                "name": "send_email",

                "url": "TOOL_SERVER_URL",

                    "type": "object",

                        "recipient": {"type": "string", "description": "Email recipient"},
