```python Python
from subconscious import Subconscious

client = Subconscious(api_key="YOUR_API_KEY")

# Block until the run finishes and return the final result.
run = client.run_and_wait(
    engine="tim",
    input={
        "instructions": "What's the derivative of 3x^2 * sin(x)?"
    },
)

print(run.result.answer)
```

```javascript JavaScript
import { Subconscious } from "subconscious";

const client = new Subconscious({
  apiKey: "YOUR_API_KEY",
});

// Block until the run finishes and return the final result.
const run = await client.runAndWait({
  engine: "tim",
  input: {
    instructions: "What's the derivative of 3x^2 * sin(x)?",
  },
});

console.log(run.result?.answer);
```
